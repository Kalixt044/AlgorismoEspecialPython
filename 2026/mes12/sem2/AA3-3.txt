## 1. Introducción y Justificación Tecnológica

### 1.1 Presentación del Diseño Final

La presente evidencia documenta la materialización de los mockups desarrollados en la EV02 en interfaces gráficas funcionales de alta fidelidad para el proyecto "Censo Rural". Este sistema integral está diseñado para optimizar la recolección, gestión y análisis de datos personales de habitantes en zonas rurales apartadas de Colombia, donde la conectividad es limitada y las condiciones operativas son desafiantes.

El diseño final implementa dos módulos principales:
- **Módulo móvil (React Native):** Aplicación offline-first para encuestadores en campo con tablets Android
- **Módulo web administrativo (HTML/CSS/JavaScript):** Plataforma para supervisores y administradores con dashboards analíticos

### 1.2 Justificación Tecnológica

#### React Native para Módulo Móvil

**Razones de selección:**

**1. Rendimiento nativo:**
- Compilación a código nativo (no webview)
- Acceso directo a APIs del dispositivo (GPS, cámara, almacenamiento)
- Fluidez en animaciones y transiciones críticas para UX

**2. Arquitectura modular:**
- Componentes reutilizables (DRY principle)
- Facilita testing unitario independiente
- Escalabilidad mediante composición de componentes

**3. Ecosistema robusto:**
- SQLite integrado para persistencia offline
- Bibliotecas maduras: React Navigation, React Native Paper
- Comunidad activa con soluciones probadas

**4. Desarrollo multiplataforma:**
- Codebase único para Android e iOS (80% compartido)
- Reduce tiempo y costo de desarrollo
- Facilita mantenimiento y actualizaciones

import DateTimePicker from '@react-native-community/datetimepicker';

const FechaNacimientoSelector = ({ value, onChange }) => {
  const [showPicker, setShowPicker] = useState(false);
  const [edad, setEdad] = useState(null);

  const calcularEdad = (fecha) => {
    const hoy = new Date();
    const nacimiento = new Date(fecha);
    let edad = hoy.getFullYear() - nacimiento.getFullYear();
    const mes = hoy.getMonth() - nacimiento.getMonth();
    if (mes < 0 || (mes === 0 && hoy.getDate() < nacimiento.getDate())) {
      edad--;
    }
    return edad;
  };

  const handleChange = (event, selectedDate) => {
    setShowPicker(false);
    if (selectedDate) {
      onChange(selectedDate);
      setEdad(calcularEdad(selectedDate));
    }
  };

  return (
    <View style={styles.fieldContainer}>
      <Text style={styles.label}>Fecha de Nacimiento *</Text>
      <TouchableOpacity 
        style={styles.dateButton} 
        onPress={() => setShowPicker(true)}
      >
        <Text>{value ? value.toLocaleDateString() : 'Seleccione fecha'}</Text>
      </TouchableOpacity>
      {edad !== null && (
        <Text style={styles.infoText}>Edad calculada: {edad} años</Text>
      )}
      {showPicker && (
        <DateTimePicker
          value={value || new Date()}
          mode="date"
          display="default"
          onChange={handleChange}
          maximumDate={new Date()}
        />
      )}

  );
};

### 3.3 Validaciones en Tiempo Real

**Sistema centralizado de validación:**

```javascript
const validarFormulario = (data) => {
  const errors = {};

  // Validar tipo y número de documento
  if (!data.tipoDocumento) {
    errors.tipoDocumento = 'Seleccione un tipo de documento';
  }
  if (!data.numeroDocumento || data.numeroDocumento.length < 6) {
    errors.numeroDocumento = 'El documento debe tener al menos 6 dígitos';
  }

  // Validar nombres
  if (!data.primerNombre || data.primerNombre.trim().length < 2) {
    errors.primerNombre = 'El primer nombre es obligatorio';
  }
  if (!data.primerApellido || data.primerApellido.trim().length < 2) {
    errors.primerApellido = 'El primer apellido es obligatorio';
  }
  if (!data.segundoApellido || data.segundoApellido.trim().length < 2) {
    errors.segundoApellido = 'El segundo apellido es obligatorio';
  }

  // Validar demografía
  if (!data.fechaNacimiento) {
    errors.fechaNacimiento = 'La fecha de nacimiento es obligatoria';
  }
  if (!data.sexo) {
    errors.sexo = 'Seleccione el sexo';
  }

  // Validar email si se proporciona
  if (data.email && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(data.email)) {
    errors.email = 'Email inválido';
  }

  // Validar celular si se proporciona
  if (data.celular && !/^\d{10}$/.test(data.celular.replace(/\D/g, ''))) {
    errors.celular = 'El celular debe tener 10 dígitos';
  }

  return errors;
};
```

### 4.1 Módulo GPS (RF-02)

**Implementación de captura de coordenadas:**

```javascript
import Geolocation from '@react-native-community/geolocation';
import { PermissionsAndroid, Platform } from 'react-native';

const ModuloGPS = ({ onCoordenadas }) => {
  const [ubicacion, setUbicacion] = useState(null);
  const [precision, setPrecision] = useState(null);
  const [cargando, setCargando] = useState(false);

  const solicitarPermisos = async () => {
    if (Platform.OS === 'android') {
      const granted = await PermissionsAndroid.request(
        PermissionsAndroid.PERMISSIONS.ACCESS_FINE_LOCATION
      );
      return granted === PermissionsAndroid.RESULTS.GRANTED;
    }
    return true;
  };

  const capturarUbicacion = async () => {
    const tienePermiso = await solicitarPermisos();
    if (!tienePermiso) {
      Alert.alert('Permiso denegado', 'Se requiere acceso a la ubicación');
      return;
    }

    setCargando(true);
    Geolocation.getCurrentPosition(
      (position) => {
        const coords = {
          latitud: position.coords.latitude,
          longitud: position.coords.longitude
        };
        setUbicacion(coords);
        setPrecision(position.coords.accuracy);
        onCoordenadas(coords);
        setCargando(false);
      },
      (error) => {
        Alert.alert('Error', 'No se pudo obtener la ubicación');
        setCargando(false);
      },
      { enableHighAccuracy: true, timeout: 20000, maximumAge: 1000 }
    );
  };

  return (
    <View style={styles.gpsContainer}>
      <Text style={styles.label}>Ubicación GPS</Text>
      <TouchableOpacity 
        style={styles.btnGPS} 
        onPress={capturarUbicacion}
        disabled={cargando}
      >
        <Text style={styles.btnText}>
          {cargando ? '📍 Capturando...' : '📍 Capturar Ubicación'}
        </Text>
      </TouchableOpacity>
      
      {ubicacion && (
        <View style={styles.coordenadasInfo}>
          <Text style={styles.coordenadas}>
            Lat: {ubicacion.latitud.toFixed(6)}
          </Text>
          <Text style={styles.coordenadas}>
            Lon: {ubicacion.longitud.toFixed(6)}
          </Text>
          {precision && (
            <Text style={[
              styles.precision,
              precision < 20 ? styles.precisionAlta : styles.precisionBaja
            ]}>
              Precisión: ±{precision.toFixed(0)}m
            </Text>
          )}
        </View>
      )}
    </View>
  );
};
```

import { launchCamera, launchImageLibrary } from 'react-native-image-picker';

const GestionArchivos = ({ onImagenSeleccionada }) => {
  const [imagenes, setImagenes] = useState([]);

  const opciones = {
    mediaType: 'photo',
    maxWidth: 1024,
    maxHeight: 1024,
    quality: 0.8
  };

  const tomarFoto = () => {
    launchCamera(opciones, (response) => {
      if (!response.didCancel && !response.error) {
        const nuevaImagen = {
          uri: response.assets[0].uri,
          nombre: response.assets[0].fileName,
          tamaño: response.assets[0].fileSize
        };
        setImagenes([...imagenes, nuevaImagen]);
        onImagenSeleccionada(nuevaImagen);
      }
    });
  };

  const seleccionarGaleria = () => {
    launchImageLibrary(opciones, (response) => {
      if (!response.didCancel && !response.error) {
        const nuevaImagen = {
          uri: response.assets[0].uri,
          nombre: response.assets[0].fileName,
          tamaño: response.assets[0].fileSize
        };
        setImagenes([...imagenes, nuevaImagen]);
        onImagenSeleccionada(nuevaImagen);
      }
    });
  };

  return (
    <View style={styles.archivoContainer}>
      <Text style={styles.label}>Fotografías</Text>
      <View style={styles.botonesArchivo}>
        <TouchableOpacity style={styles.btnArchivo} onPress={tomarFoto}>

const calcularContraste = (color1, color2) => {
  const luminancia = (r, g, b) => {
    const [rs, gs, bs] = [r, g, b].map(c => {
      c = c / 255;
      return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);

};

            styles.pasoCirculo,
            pasoActual >= paso.numero && styles.pasoActivo,
            pasoActual > paso.numero && styles.pasoCompletado
          ]}>

          {index < pasos.length - 1 && (
            <View style={[
              styles.pasoLinea,
              pasoActual > paso.numero && styles.lineaCompletada
            ]} />
          )}
        </View>
      ))}
    </View>
  );
