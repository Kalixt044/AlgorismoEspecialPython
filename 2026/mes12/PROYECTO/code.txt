    // Máxima fecha = hoy para fecha de nacimiento
    const fechaNac = document.getElementById('fechaNac');
    const edad = document.getElementById('edad');
    const statusEl = document.getElementById('form-status');

    (function setMaxDateToday(){
      const today = new Date();
      const y = today.getFullYear();
      const m = String(today.getMonth()+1).padStart(2,'0');
      const d = String(today.getDate()).padStart(2,'0');
      const iso = `${y}-${m}-${d}`;
      if (fechaNac) fechaNac.max = iso;
    })();

    // Cálculo de edad
    function calcularEdad(fechaISO){
      const hoy = new Date();
      const nac = new Date(fechaISO);
      let e = hoy.getFullYear() - nac.getFullYear();
      const m = hoy.getMonth() - nac.getMonth();
      if (m < 0 || (m === 0 && hoy.getDate() < nac.getDate())) e--;
      return e;
    }
    fechaNac?.addEventListener('change', () => {
      if (fechaNac.value) {
        const e = calcularEdad(fechaNac.value);
        edad.value = isFinite(e) && e >= 0 ? e : '';
      } else {
        edad.value = '';
      }
    });

    // Validación básica y mensaje de estado
    const form = document.getElementById('form-persona');
    form?.addEventListener('submit', (ev) => {
      statusEl.textContent = '';
      const ok = form.checkValidity();
      if (!ok) {
        ev.preventDefault();
        form.reportValidity();
        statusEl.innerHTML = 'Por favor, completa los campos obligatorios marcados con *.';
        statusEl.classList.add('error');
      } else {
        // Aquí podrías enviar con fetch() a tu backend
        statusEl.textContent = 'Formulario enviado correctamente.';
        statusEl.classList.remove('error');
      }
    });
  </script>
